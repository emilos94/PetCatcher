#version 330 core
out vec4 FragColor;

in vec3 pass_normal;
in vec2 pass_uv;

uniform vec3 color;
uniform float use_color;
uniform float use_texture;
uniform sampler2D texture_sampler;

uniform float ambient_strength;
uniform vec3 light_color;

void main() {
    vec3 ambient = ambient_strength * light_color;

    vec4 object_color = vec4(color * use_color, 1.0) + texture(texture_sampler, pass_uv) * use_texture;

    FragColor = object_color * vec4(ambient, 1.0);
}
